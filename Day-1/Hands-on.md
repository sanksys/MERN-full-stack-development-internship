#Git hands-on
1. git config --global user.name "Your Name"
    run only once on system
2. git config --global user.email "your_email@example.com"
    run only once on system
3. git init
    run only once per project/repository
4. git status
5. git add .
6. git commit -m 'Commit message'
7. git branch -M main 
    run only once per project/repository
8. git remote add origin https://github.com/sanksys/MERN-full-stack-development-internship.git
    run only once per project/repository
    change the url to repository url
9. git push -u origin main
